.\bin\windows\kafka-server-start.bat .\config\kraft\server.properties

# Main topic
.\bin\windows\kafka-topics.bat --create --topic employee --bootstrap-server localhost:9092 --partitions 3 --replication-factor 1

# Retry Topics
.\bin\windows\kafka-topics.bat --create --topic employee-retry-0 --bootstrap-server localhost:9092 --partitions 3 --replication-factor 1
.\bin\windows\kafka-topics.bat --create --topic employee-retry-1 --bootstrap-server localhost:9092 --partitions 3 --replication-factor 1
.\bin\windows\kafka-topics.bat --create --topic employee-retry-2 --bootstrap-server localhost:9092 --partitions 3 --replication-factor 1

# Dlt Topic
.\bin\windows\kafka-topics.bat --create --topic employee-dlt --bootstrap-server localhost:9092 --partitions 3 --replication-factor 1


# List all the topics
 .\bin\windows\kafka-topics.bat --list --bootstrap-server localhost:9092
